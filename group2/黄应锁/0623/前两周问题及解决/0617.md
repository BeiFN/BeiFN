        var res= {};
        for(var i in arr){
            var c = arr[i]; //c为对象中的key值；
            if(res[c]){
                res[c] += 1;
            }else{
                res[c] = 1;
            }
        }

          console.log("字符串"+j+"出现的次数："+res[j]+"次");

        }
    </script>
