5. ```javascript
   var a = 10; // 全局变量 a;  window.a = 10;
   var obj = {
       a: 20,  // obj.a == 20;
       b: function () {
           console.log(this) // obj;
           setTimeout(function () {
               console.log(this.a); // window   ==> window.a
           }, 0)
       },
       c: function () {
           console.log(this.a);
       }
   }
   obj.b(); //10;
   obj.c(); //20;
   为什么？
   ```

   > 就近原则;   =>  this指向只看最近的一个函数;
