* 函数去抖 

无论你执行多少次,我都在最后一次执行延时n ms之后执行。

```javascript
      // 函数高频次执行 持续5s ，实际我需要的渲染次数只有一次;
      function foo(){}
      let timer = setInterval( foo , 1 );
      setTimeout(()=> {
            clearInterval(timer)
      } , 5000)

      // 如果在一定时间之内还有下一次执行, 那么取消本次执行;
      function debounce( cb , delay ){
            let timer = null;
            return function(){
                  clearInterval(timer);
                  timer = setInterval(()=> {
                        cb();
                  },delay)
            }
      }

      //更新;
      function foo(){}
      let timer = setInterval( debounce( foo , 200 ) , 1 );
      setTimeout(()=> {
            clearInterval(timer)
      } , 5000)
```
