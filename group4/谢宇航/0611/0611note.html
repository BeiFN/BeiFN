5.do while循环
            do{
                  // 无论如何都会先执行do里面代码;
                  console.log("do ...");      
                  // 如果while 条件为true ,那么 会循环执行do之中代码;
            }while(false){
                  // 一旦while循环条件为false,那么while代码块就执行了;
                  console.log("while ...");
            }
            do while 比 while 多执行一次;
            多一个循环终止功能;

6.for循环
            for 循环 ; 是高级循环封装;


            for(声明变量;循环逻辑;自增自减用){


            }


            for(var i = 0 ; i < 10 ; i ++){
                  console.log(i);
            }

                  console.log(i);
            }
