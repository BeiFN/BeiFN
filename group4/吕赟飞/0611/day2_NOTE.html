switch
          
   switch(Number(month)){
                  case 1:
                  case 3:
                  case 5:
                  case 7:
                  case 8:
                  case 10:
                  case 12:
                        alert(month + "月有31天");
                        break;
                  case 4 :
                  case 6 :
                  case 9 :
                  case 11 :
                        alert(month + "月有30天");
                        break;
                  case 2 :
                        if(year % 400 === 0 || year % 4 === 0 && year % 100 !== 0){
                              alert(month + "月有29天");
                        }else{
                              alert(month + "月有28天");
                        }
                        break;
                  default :
                        alert("条件不合法");
            }


            特性1 :
            穿透特性; 如果switch条件满足了某个case，然后我们的switch会继续向下执行下面所有的条件代码;

            阻止穿透 : break;

            什么情况下选择switch 而不选择 else if;  条件必须是具体值 , 而不能是范围。 推荐使用 switch ;

            特性2:
            小括号里的条件, 和case的比对是绝对比对 === ,如果数据类型不等那么不会继续比对了;

打印1 ~ 100 之间的素数;
            1. 反证法;
            2. 标识变量;
            3. break ; continue ;

            var count = 1;
            while(count ++ < 100){
                  var flag_count = true ; // 是不是素数; 发起一个假设，假设这个count 是素数;
                  // 判定 count 是不是素数; => 找到count不是素数的依据;
                  // console.log("----------" + count + "-----------")
                  var num = 1;
                  while(num ++ < count - 1){
                        // num 是 : 除了 1 和 count 本身之中的所有可能整除这个数的数字;
                        if(count % num === 0){
                              // 不是素数;
                              // tip : 只要进入了这个逻辑 , 那么 count 就一定不是素数;
                              // 如果进入了这里那么逻辑记录为false;
                              flag_count = false;
                              break;
                        }
                  }
                  // 如果经过了，上面的循环，没有进入 if 逻辑那么这个数就是素数;
                  if(flag_count){
                        console.log(count);
                  }
            }

                  var hen = -1;

                        chick = 100 - cock - hen;
                        if(cock * 5 + hen* 3 + chick / 3 === 100){
                              console.log(cock,hen,chick);
                        }
                  }
            }
